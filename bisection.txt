clc;
clear all
F = @(x) x^2 - 29;
i = 1;
a = 5 ;
tol = 10^-5;
b = 6 ;
N = 20;
while i<N
    c = (a+b)/2;
    if F(a)*F(c)< 0
        b = c;
    end        
    if F(b)*F(c)< 0
        a = c;
    end
    if abs(a-b)<tol
       fprintf("The root is %f",c);
       break;
    end
    i = i+1;
end
if i == N
    fprintf("root not found,change interval");
end
